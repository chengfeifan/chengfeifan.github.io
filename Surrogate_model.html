# define a function to calculate CDM

CDMCalc<-function(x){
  CDMValue<-lapply(x,function(x_i){
    xi<-rep(x_i,length(x))
    return(sum((xi-x)^2))
  })
  return(unlist(CDMValue))
}

# Calulate CDM
CDMValue<-CDMCalc(x)
orderCDM<-order(CDMValue,decreasing = TRUE)
